def generate_row(row)
    @data.length != @data.index(row) + 1 ? 
      "<tr>
        #{(row.map {|x| "<td>#{generate_item(x)}</td>"}).join}
      </tr>\n" :
      "<tr>
        #{(row.map {|x| "<td>#{generate_item(x)}</td>"}).join}
      </tr>\n</table>\n"
end
